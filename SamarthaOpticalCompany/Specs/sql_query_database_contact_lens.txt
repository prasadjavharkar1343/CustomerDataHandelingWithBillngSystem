========================================specs========================================

CREATE TABLE `specs` (
 `id` int(5) NOT NULL AUTO_INCREMENT,
 `name` varchar(100) NOT NULL,
 `age` int(3) NOT NULL,
 `gender` varchar(100) NOT NULL,
 `address` varchar(100) NOT NULL,
 `contact` int(12) NOT NULL,
 `model` varchar(100) NOT NULL,
 `visitingdate` date NOT NULL,
 `mrp` double NOT NULL,
 `file` text NOT NULL,
 `right_sph1` double DEFAULT NULL,
 `right_cyl1` double DEFAULT NULL,
 `right_axis1` double DEFAULT NULL,
 `right_vn1` varchar(10) DEFAULT NULL,
 `right_sph2` double DEFAULT NULL,
 `right_cyl2` double DEFAULT NULL,
 `right_axis2` double DEFAULT NULL,
 `right_vn2` varchar(10) DEFAULT NULL,
 `left_sph1` double DEFAULT NULL,
 `left_cyl1` double DEFAULT NULL,
 `left_axis1` double DEFAULT NULL,
 `left_vn1` varchar(10) DEFAULT NULL,
 `left_sph2` double DEFAULT NULL,
 `left_cyl2` double DEFAULT NULL,
 `left_axis2` double DEFAULT NULL,
 `left_vn2` varchar(10) DEFAULT NULL,
 PRIMARY KEY (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=18 DEFAULT CHARSET=latin1
